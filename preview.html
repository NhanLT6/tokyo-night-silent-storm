/// <summary>Processes customer orders.</summary>
[Service]
public class OrderService : IOrderService
{
    // Delegate type
    public delegate void OrderHandler(Order order);

    // Event
    public event OrderHandler OrderPlaced;

    private readonly ILogger<OrderService> _logger;

    public OrderService(ILogger<OrderService> logger)
    {
        _logger = logger;
    }

    // Generic method with type parameter
    public T GetConfig<T>(string key) =>
        ConfigReader.Get<T>(key);

    // Async method — format string, mutable local, null-coalescing
    public async Task<OrderResult> PlaceOrderAsync(Order order)
    {
        var id = order.Id ?? Guid.NewGuid();
        _logger.LogInformation("Placing order {0} for {1}", id, order.Customer);

        // Inlay hints (LINQ chain) — dimmed so they don't fight the code
        var results = order.Items
            .Where(item: x => x.Stock > 0)
            .Select(selector: x => x.Price * x.Qty)
            .ToList();

        await SaveAsync(order);
        OrderPlaced?.Invoke(order);
        return new OrderResult { Id = id };
    }
}
